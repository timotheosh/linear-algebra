import numpy as np
import sys
import traceback
#sys.stderr = sys.stdout

def format_value(arg):
    try:
        print(arg.tolist())
    except:
        print(traceback.format_exc())
        sys.stdout.flush()
#+end_src

#+begin_src jupyter-python :session vectors :results none
